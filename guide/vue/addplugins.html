main.js入口文件

import Vue from 'vue'
import App from './App.vue'
import '@/assets/scss/main.scss'
import ElementUI from 'element-ui';
Vue.use(ElementUI);

//亦云组件库
import yiyunMessage from '@/components/message/index.js';
import yiyunLoading from '@/components/loading/index.js';

console.log('Yiyun components have been loaded successfully')
const components = [
  yiyunMessage,
  yiyunLoading,
]

//暴露install方法
const install = function (Vue) {
  if (install.installed) return
  install.installed = true
  // 遍历并注册全局组件
  components.forEach(component => {
    Vue.component(component.name, component)
  })
  Vue.prototype.$yiyunMessage = yiyunMessage;
  Vue.use(yiyunLoading)
}
if (typeof window !== 'undefined' && window.Vue) {
  install(window.Vue)
}
const yiyunComponents = {
  // 导出的对象必须具备一个 install 方法
  version: '1.0.1',
  install,
  // 组件列表
  yiyunInput,
  yiyunEmpty,
  yiyunCustomForm,
  yiyunOrganization,
  yiyunOrganizationSource,
  yiyunLevelModel,
  yiyunMessage,
  yiyunLoading,
}
export default yiyunComponents

//挂载app 本地模拟引入插件库
if (process.env.NODE_ENV === "development") {
  Vue.use(yiyunComponents)
  Vue.prototype.$yiyunMessage = yiyunMessage;
  Vue.use(yiyunLoading)
  Vue.config.productionTip = false
  new Vue({
    render: h => h(App),
  }).$mount('#app')
}

main.ts入口文件

import { createApp } from 'vue'
import App from './App.vue'
import router from './router'
import store from './store'
import { ElTree } from 'element-plus';
import '@/assets/scss/main.scss'

const app = createApp(App)

//核心：
//1.引入封装好的插件
import YiyunOrganization from './components/yiyunOrganization'
const components = [YiyunOrganization]
//2.暴露install()方法的Object
const install = (app:any, opt:any):void => {
    app.component(ElTree.name, ElTree);
    components.forEach((component:any) => {
        app.component(component.name, component)
    });
};
const version = "1.1.5";
console.log(`yiyun components ${version} has loaded successfully !`)
var index = {
    version,
    install,
};

//挂载app 本地模拟引入插件库
if(process.env.NODE_ENV!=="production") {
    app.use(index)
    app.use(store).use(router).mount('#app')
}
export default index;

发布到npm

  • 1.如果以前改过 npm 的镜像地址，比如使用了淘宝镜像，就先改回来
npm config set registry http://registry.npmjs.org
1
  • 2.如果没有 npm 账户，先创建一个账户，或者到 npm 官网注册
npm adduser
1
  • 3.如果在官网注册的账户，或者以前就有账户 则进行登录
npm login
1
  • 4.测试是否已经登录
npm whoami
1
  • 5.最后发布包
npm publish
1

具体流程可以参考官方文档如何发布程序包

TIP

package.json文件下的files字段设置为[]，可以快速过滤不想发布到npm服务器上的文件

{
  "files":[] //只发布入口文件
}
