  data() {
    return {
      page: 1,
      num: 10,

      tableData: [],

  async created() {
    try {
      this.$utils.showLoading();
      await this.getList();
      this.$utils.hideLoading();
    } catch (error) {
      this.$utils.hideLoading();
    }
  },
  methods: {
    //获取列表
    async getList() {
      const res = await this.ServerApi({
        page: this.page,
        num: this.num,
      });
      this.total = res.total;
      this.tableData = res.data;
    },
    //搜索
    async confirmFilter() {
      try {
        this.$utils.showLoading();
        this.page = 1;
        await this.getList();
        this.$utils.hideLoading();
      } catch (error) {
        this.$utils.hideLoading();
      }
    },
    //修改每页条数
    async changePageSize(num) {
      try {
        this.$utils.showLoading();
        this.page = 1;
        this.num = num;
        await this.getList();
        this.$utils.hideLoading();
      } catch (error) {
        this.$utils.hideLoading();
      }
    },
    //切页

      try {
        this.$utils.showLoading();
        this.page = page;
        await this.getList();
        this.$utils.hideLoading();
      } catch (error) {
        this.$utils.hideLoading();
      }
    },
};

export default {
  data() {
    return {
      organizationVisible: false,
      departmentList: [],
      memberList: [],
      checkedMembers: [],
      memberName:"",
      memberId:""
    };
  },
  async created() {
    await this.getOrganizationData();
  },
  methods: {
    ...mapActions(["GetDeptList", "GetMemberInfoByDept"]),
    async getOrganizationData() {
      const deptInfo = await this.GetDeptList();
      this.departmentList = deptInfo.data;
      const deptIds = [...deptInfo.data.map((dept) => dept.id), ...["-1"]];
      const memberInfo = await this.GetMemberInfoByDept({
        page: 1,
        num: 1000,
        isAll: true,
        memberType: 0,
        deptIdList: deptIds,
      });
      this.memberList = memberInfo.data;
    },
    openOrganization() {
      this.checkedMembers = this.memberId?[{id:this.memberId,name:this.memberName}]:[]
      this.organizationVisible = true;
    },
    confirmOrganization(checkResult) {
      console.log("选择结果", checkResult);
      this.memberId = checkResult.checkedList.length?checkResult.checkedList[0].id:"";
      this.memberName = checkResult.checkedList.length?checkResult.checkedList[0].name:"";
      this.organizationVisible = false;
    },
    closeOrganization() {
      this.organizationVisible = false;
    },
  },
};

export default {
  data() {
    return {
      startTime: "",
      endTime: "",
    };
  },
  methods: {
    //禁用大于结束时间的日期
    disabledStartTime(currentDate) {
      if (this.endTime) {
        return new Date(currentDate).getTime() >= this.endTime;
      }
    },
    //禁用小于开始时间的日期
    disabledEndTime(currentDate) {
      if (this.startTime) {
        return new Date(currentDate).getTime() < this.startTime;
      }
    },
    changeStartTime(timestamp) {
      this.startTime = timestamp;
    },
    clearStartTime() {
      this.startTime = "";
    },
    changeEndTime(timestamp) {
      this.endTime = timestamp;
    },
    clearEndTime() {
      this.endTime = "";
    },
  }
};

export default {
  data() {
    return {
      ossImage:"",
    };
  },
  methods: {
    async uploadFile(file) {
      try {
        console.log(file);
        const fileType = this.$utils.getFileType(file.name);
        if (!["jpg", "gif", "png"].includes(fileType)) {
          this.$message.error("图片格式为：JPG/GIF/PNG");
          return;
        }
        if (file.size / 1024 / 1024 > 5) {
          this.$message.error("图片不能超过5M");
          return;
        }
        const res = await this.$compressImage(file.raw);
        this.$utils.showLoading();
        let ossUrl = await this.uploadOss({
          file: res.file,
        });
        console.log(ossUrl);
        this.ossImage = ossUrl
        this.$utils.hideLoading();
      } catch (error) {
        this.$message.error(error);
        this.$utils.hideLoading();
      }
    }
  }
};

export default {
  data() {
    return {
      excelData:[],
    };
  },
  methods: {
    requestImageToBase64(imgUrl) {
      return new Promise((resolve, reject) => {
        const xhr = new XMLHttpRequest();
        xhr.open("get", imgUrl, true);
        xhr.responseType = "blob";
        xhr.onload = async (e) => {
          if (e.target.status == 200) {
            const blob = e.target.response;
            const res = await this.$compressImage(blob);
            resolve(res.base64);
          } else {
            resolve(imgUrl);
          }
        };
        xhr.send();
      });
    },
    async exportExcel() {
      if (!this.excelData.length) {
        this.$message.error("请选择需要导出的兑换记录！");
        return;
      }
      try {
        this.$utils.showLoading();
        const grids = []
        const excelData_clone = JSON.parse(JSON.stringify(this.excelData));
        for (let cell of excelData_clone) {
          const base64 = await this.requestImageToBase64(cell.mainPic);
          cell.mainPic = base64;
          const cellRow = [
            { value: cell.mainPic, isImage: true },
            { value: cell.title },
            { value: cell.orderNumber },
            { value: this.statusEnmu[cell.status] },
            { value: cell.qty },
            { value: cell.totalPrice },
            { value: cell.buyerName },
            {
              value: this.$_moment(cell.limitTime * 1000).format(
                "YYYY/MM/DD HH:mm"
              ),
            },
            {
              value: this.$_moment(cell.createTime * 1000).format(
                "YYYY/MM/DD HH:mm"
              ),
            },
          ];
          grids.push(cellRow)
        }
        grids.unshift([
          { value: "商品图片" },
          { value: "商品名称" },
          { value: "订单编号" },
          { value: "状态" },
          { value: "数量" },
          { value: "总价" },
          { value: "兑换人" },
          { value: "商品兑换期限" },
          { value: "下单时间" },
        ]);
        await this.$exportExcelGrid({
          fileName: "兑换记录",
          rowHeights: [],
          colWidths: [80, 360, 280],
          grids: grids,
          merges: [],
          exportImmediately: true,
        });
        this.$message.success("导出成功");
        this.$utils.hideLoading();
      } catch (error) {
        this.$utils.hideLoading();
      }
    },
  }
};
