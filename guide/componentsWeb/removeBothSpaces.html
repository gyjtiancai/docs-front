去除输入框两边空格

调用方式

//源生dom  
<input type="text" v-removeBothSpaces />
//ui组件
<el-input v-removeBothSpaces v-model="name"></el-input>
//元素包裹
<div v-removeBothSpaces>
    <el-input v-model="name2"></el-input>
    <el-input v-model="name1"></el-input>
    <input type="text" />
    <input type="text" />
    <input type="text" />
</div>

    <el-dialog title="收货地址" :visible.sync="dialogTableVisible">
        <div>
            <el-input type="text" v-model="name4"></el-input>
            <input type="text" v-model="name3" />
        </div>
    </el-dialog>
</div>
